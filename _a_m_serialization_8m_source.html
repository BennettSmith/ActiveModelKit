00027 // Import Objective-C Message Header
00028 // When using Automatic Reference Counting, avoid using -[NSObject
00029 // performSelector:] to circumvent the "may cause a leak because its selector is
00030 // unknown" warning. Instead use objc_msgSend, but this requires Objective-C
00031 // run-time headers.
00032 #import <objc/message.h>
00033 
00034 NSString *const kAMOnlyOptionKey = @"only";
00035 NSString *const kAMExceptOptionKey = @"except";
00036 NSString *const kAMMethodsOptionKey = @"methods";
00037 
00038 // Rails implements Active Model serialisation using mix-ins. Objective-C has
00039 // nothing similar, let alone equivalent. Protocols are the only thing you can
00040 // ‘mix in,’ so to speak. However, Objective-C protocols cannot implement
00041 // behaviour but can only declare interfaces. Hence wrap the behaviour as a
00042 // functional helper.
00043 NSDictionary *AMSerializableHash(id<AMAttributeMethods> objectWithAttributes, NSDictionary *options)
00044 {
00045     NSMutableDictionary *hash = [NSMutableDictionary dictionaryWithDictionary:[objectWithAttributes attributes]];
00046     
00047     NSArray *only = [options objectForKey:kAMOnlyOptionKey];
00048     if (only)
00049     {
00050         NSMutableSet *attributeNames = [NSMutableSet setWithArray:[hash allKeys]];
00051         [attributeNames minusSet:[NSSet setWithArray:only]];
00052         [hash removeObjectsForKeys:[attributeNames allObjects]];
00053     }
00054     else
00055     {
00056         NSArray *except = [options objectForKey:kAMExceptOptionKey];
00057         if (except)
00058         {
00059             [hash removeObjectsForKeys:except];
00060         }
00061     }
00062     
00063     for (NSString *methodName in [options objectForKey:kAMMethodsOptionKey])
00064     {
00065         SEL selector = NSSelectorFromString(methodName);
00066         if ([objectWithAttributes respondsToSelector:selector])
00067         {
00068             [hash setObject:objc_msgSend(objectWithAttributes, selector) forKey:methodName];
00069         }
00070     }
00071     
00072     // Unlike the Rails counterpart, do not send the attribute names as
00073     // getters. Optimise by assuming that the original message (above) to access
00074     // all attributes along with attribute names as keys constitutes the getting
00075     // phase for attributes.
00076     return [hash copy];
00077 }
