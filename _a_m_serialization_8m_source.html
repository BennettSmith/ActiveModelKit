27 // Import Objective-C Message Header
28 // When using Automatic Reference Counting, avoid using -[NSObject
29 // performSelector:] to circumvent the "may cause a leak because its selector is
30 // unknown" warning. Instead use objc_msgSend, but this requires Objective-C
31 // run-time headers.
32 #import <objc/message.h>
33 
34 NSString *const kAMOnlyOptionKey = @"only";
35 NSString *const kAMExceptOptionKey = @"except";
36 NSString *const kAMMethodsOptionKey = @"methods";
37 
38 // Rails implements Active Model serialisation using mix-ins. Objective-C has
39 // nothing similar, let alone equivalent. Protocols are the only thing you can
40 // ‘mix in,’ so to speak. However, Objective-C protocols cannot implement
41 // behaviour but can only declare interfaces. Hence wrap the behaviour as a
42 // functional helper.
43 NSDictionary *AMSerializableHash(id<AMAttributeMethods> objectWithAttributes, NSDictionary *options)
44 {
45  NSMutableDictionary *hash = [NSMutableDictionary dictionaryWithDictionary:[objectWithAttributes attributes]];
46 
47  NSArray *only = [options objectForKey:kAMOnlyOptionKey];
48  if (only)
49  {
50  NSMutableSet *attributeNames = [NSMutableSet setWithArray:[hash allKeys]];
51  [attributeNames minusSet:[NSSet setWithArray:only]];
52  [hash removeObjectsForKeys:[attributeNames allObjects]];
53  }
54  else
55  {
56  NSArray *except = [options objectForKey:kAMExceptOptionKey];
57  if (except)
58  {
59  [hash removeObjectsForKeys:except];
60  }
61  }
62 
63  for (NSString *methodName in [options objectForKey:kAMMethodsOptionKey])
64  {
65  SEL selector = NSSelectorFromString(methodName);
66  if ([objectWithAttributes respondsToSelector:selector])
67  {
68  [hash setObject:objc_msgSend(objectWithAttributes, selector) forKey:methodName];
69  }
70  }
71 
72  // Unlike the Rails counterpart, do not send the attribute names as
73  // getters. Optimise by assuming that the original message (above) to access
74  // all attributes along with attribute names as keys constitutes the getting
75  // phase for attributes.
76  return [hash copy];
77 }
